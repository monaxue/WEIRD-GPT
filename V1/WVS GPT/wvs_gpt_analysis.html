data_cleaned <- data %>%
  mutate_all(~ case_when(. < 0 ~ NA, TRUE ~ .)) %>%
  group_by(B_COUNTRY) %>%
  mutate_at(vars(-c(B_COUNTRY, starts_with(c("core_3", "security_3", "security_5", "security_9", "religion_2", "religion_7", "religion_8", "trust_6", "econ_values_2", "security_7", "security_8", "trust_5", "postmat_1", "postmat_2", "postmat_3")))), ~mean(., na.rm = T)) %>%
